import arcade import math WIDTH = 1000 HEIGHT = 700 class LoveStory(arcade.Window): def __init__(self): super().__init__(WIDTH, HEIGHT, "❤️ Для тебя ❤️") self.timer = 0 self.messages = [ (2, "Привет ❤️"), (5, "Я написал эту программу специально для тебя"), (8, "Потому что ты очень важна для меня"), (11, "И я хочу, чтобы ты улыбнулась 😊"), (14, "Спасибо, что ты есть в моей жизни ❤️") ] def on_update(self, delta_time): self.timer += delta_time def draw_heart(self): scale = 12 + math.sin(self.timer * 4) * 1.5 points = [] for t in range(360): angle = math.radians(t) x = 16 * math.sin(angle) ** 3 y = ( 13 * math.cos(angle) - 5 * math.cos(2 * angle) - 2 * math.cos(3 * angle) - math.cos(4 * angle) ) x = x * scale + WIDTH / 2 y = y * scale + HEIGHT / 2 - 50 points.append((x, y)) arcade.draw_polygon_filled(points, arcade.color.RED) def on_draw(self): self.clear() arcade.set_background_color((15, 15, 30)) visible_text = "" for time_needed, text in self.messages: if self.timer >= time_needed: visible_text = text arcade.draw_text( visible_text, WIDTH / 2, 120, arcade.color.WHITE, 28, anchor_x="center" ) if self.timer >= 14: self.draw_heart() arcade.draw_text( "❤️ Я люблю тебя ❤️", WIDTH / 2, 500, arcade.color.PINK, 36, anchor_x="center" ) window = LoveStory() arcade.run()
